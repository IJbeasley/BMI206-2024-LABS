# ANSWER 7.1
test_pval = vector()
test_effect_size = vector()

for(gene_id in 1:ncol(mut)){
  
  with_mut = mut[, gene_id]
  
  test_res =
    wilcox.test(x = sv[with_mut, "deletions"],
              y = sv[!with_mut, "deletions"],
                  alternative = "greater",
                   conf.int = T)
